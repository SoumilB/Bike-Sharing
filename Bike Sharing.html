#Grouping the hours by 6 hours (Group1)
conditions1 = [(bike_data['hour'] >= 0) & (bike_data['hour'] < 6),
                (bike_data['hour'] >= 6) & (bike_data['hour'] < 12),
                (bike_data['hour'] >= 12) & (bike_data['hour'] < 18),
                (bike_data['hour'] >= 18) & (bike_data['hour'] < 24)]
